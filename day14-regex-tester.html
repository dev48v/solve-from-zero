🔨 Build it yourself

    
Read the pattern and flags from the inputs, compile them with new RegExp inside a try/catch, run matchAll over the test text, then paint highlights and list every capture group. ~50 lines for a real live regex tester.

    
    
      
  1. 1
    Read the pattern and gather the flags
    const pattern = patternEl.value;
// collect the checked flag boxes into one string like "gi"
const flags = [...document.querySelectorAll("#flags input:checked")]
  .map(box => box.dataset.flag)
  .join("");
// flags order doesn't matter to the engine; "gi" === "ig"
    2. 

      
  2. 2
    Compile it safely with try/catch
    let re = null, errorMsg = "";
try {
  // an invalid pattern (e.g. "(" ) THROWS a SyntaxError here
  re = new RegExp(pattern, flags);
} catch (err) {
  errorMsg = err.message;   // show this to the user, don't crash
}
if (!pattern) re = null;     // treat an empty pattern as "nothing to match"
    3. 

      
  3. 3
    Collect every match — guard zero-width loops
    const text = testEl.value;
const matches = [];
if (re && re.global) {
  let m;
  re.lastIndex = 0;
  while ((m = re.exec(text)) !== null) {
    matches.push(m);
    // CRITICAL: a zero-width match (e.g. /a*/g) never advances
    // lastIndex on its own → infinite loop. Nudge it forward.
    if (m.index === re.lastIndex) re.lastIndex++;
  }
} else if (re) {
  const m = re.exec(text);      // no /g → at most one match
  if (m) matches.push(m);
}
    4. 

      
  4. 4
    Paint highlights on a backdrop layer
    // build the test text with <mark> around each match span,
// walking left-to-right so indexes stay correct
let html = "", last = 0;
matches.forEach((m, k) => {
  const start = m.index, end = start + m[0].length;
  html += escapeHtml(text.slice(last, start));
  const cls = k % 2 ? "match alt" : "match";   // alternate colours
  html += `<mark class="${cls}">${escapeHtml(m[0]) || "∅"}</mark>`;
  last = end;
});
html += escapeHtml(text.slice(last));
backdropEl.innerHTML = html;   // sits behind a transparent textarea
    5. 

      
  5. 5
    List each match with its capture groups
    matches.forEach((m, k) => {
  // m[0] = full match; m[1], m[2]… = numbered capture groups
  const groups = m.slice(1).map((g, gi) =>
    `#${gi+1}: ${g === undefined ? "(no match)" : g}`);
  // named groups from (?<year>…) live on m.groups
  if (m.groups) for (const name in m.groups)
    groups.push(`${name}: ${m.groups[name]}`);
  console.log(`match ${k+1} at index ${m.index}: "${m[0]}"`, groups);
});
    6. 

      
  6. 6
    Show errors instead of crashing
    if (errorMsg) {
  errorEl.textContent = "⚠ Invalid regex: " + errorMsg;
  errorEl.classList.remove("hidden");
  countEl.textContent = "—";
} else {
  errorEl.classList.add("hidden");
  countEl.textContent = matches.length + " matches";
}
// a friendly message beats a blank screen and a console error

  7. 7
    Wire up presets
    const PRESETS = {
  Email: { pattern: "[\\w.+-]+@[\\w-]+\\.[\\w.-]+", flags: "gi" },
  Date:  { pattern: "(?<year>\\d{4})-(\\d{2})-(\\d{2})", flags: "g" },
  Hex:   { pattern: "#[0-9a-f]{3,6}\\b", flags: "gi" },
};
presetBtn.onclick = () => {
  patternEl.value = PRESETS.Date.pattern;
  setFlags(PRESETS.Date.flags);   // tick the right checkboxes
  render();                       // re-run immediately
};

  8. 8
    Run it live on every change
    patternEl.addEventListener("input", render);
testEl.addEventListener("input", render);
document.querySelectorAll("#flags input")
  .forEach(box => box.addEventListener("change", render));
// keep the backdrop scrolled in sync with the textarea
testEl.addEventListener("scroll", () => {
  backdropEl.scrollTop = testEl.scrollTop;
});
render();   // first paint on load
