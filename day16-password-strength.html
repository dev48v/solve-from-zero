Type a password — see how strong it is, and how long to crack it

🔨 Build it yourself

    
Detect which character sets are used → estimate entropy from length and pool size → dock bits for blocklisted words and predictable patterns → map the bits to a 0–4 score and a crack-time → render a bar and a few honest tips. ~70 lines for a real, browser-only strength meter.

    
    
      
  1. 1
    Detect the character pool
    // which kinds of characters appear? each adds to the "pool"
function poolSize(pw){
  let pool = 0;
  if (/[a-z]/.test(pw)) pool += 26;   // lowercase
  if (/[A-Z]/.test(pw)) pool += 26;   // uppercase
  if (/[0-9]/.test(pw)) pool += 10;   // digits
  if (/[^a-zA-Z0-9]/.test(pw)) pool += 33; // symbols/space
  return pool;   // e.g. lower+digits = 36 possible chars per slot
}
    2. 

      
  2. 2
    Estimate entropy in bits
    // entropy = how many bits of unpredictability.
// each slot can be 1 of `pool` chars → log2(pool) bits per char.
function baseEntropy(pw){
  const pool = poolSize(pw);
  if (pool === 0) return 0;
  return pw.length * Math.log2(pool);   // bits ≈ length × log2(pool)
}
// "abc" over 26 letters = 3 × log2(26) ≈ 14.1 bits
// LENGTH multiplies, so longer beats fancier every time

  3. 3
    Dock bits for common passwords (blocklist)
    // attackers try known passwords FIRST — these have ~0 real entropy
const BLOCKLIST = ["password","123456","12345678","qwerty","admin",
  "letmein","welcome","monkey","dragon","iloveyou","abc123","111111"];

function blocklistPenalty(pw){
  const low = pw.toLowerCase();
  for (const bad of BLOCKLIST){
    if (low === bad) return 1e9;          // exact match → crush the score
    if (low.includes(bad)) return 20;     // contains one → big dock
  }
  return 0;
}
    4. 

      
  4. 4
    Dock bits for patterns (repeats, sequences, keyboard walks)
    const ROWS = ["qwertyuiop","asdfghjkl","zxcvbnm","1234567890"];

function patternPenalty(pw){
  let p = 0;
  const low = pw.toLowerCase();
  if (/(.)\1{2,}/.test(pw)) p += 10;          // aaa, 1111 — repeats
  // sequential runs: abc, 321, 1234
  for (let i = 0; i < low.length - 2; i++){
    const a = low.charCodeAt(i), b = low.charCodeAt(i+1), c = low.charCodeAt(i+2);
    if (b - a === 1 && c - b === 1) p += 4;   // ascending run
    if (a - b === 1 && b - c === 1) p += 4;   // descending run
  }
  // keyboard walks: substring of a keyboard row
  for (const row of ROWS)
    for (let i = 0; i < low.length - 2; i++)
      if (row.includes(low.substr(i,3))) p += 6;
  return p;
}
    5. 

      
  5. 5
    Final entropy → a 0–4 score
    function strength(pw){
  // start from the length×log2(pool) estimate, subtract the penalties
  let bits = baseEntropy(pw) - blocklistPenalty(pw) - patternPenalty(pw);
  if (bits < 0) bits = 0;
  // map bits to 5 buckets: Very weak → Very strong
  let score = 0;
  if (bits >= 28) score = 1;   // weak
  if (bits >= 40) score = 2;   // fair
  if (bits >= 60) score = 3;   // strong
  if (bits >= 80) score = 4;   // very strong
  return { bits, score };
}

  6. 6
    Bits → guesses → human crack time
    // total guesses to exhaust ≈ 2^bits; on average crack at half that.
// assume a fast offline attacker: ~1e10 (10 billion) guesses/second.
function crackTime(bits){
  const seconds = Math.pow(2, bits) / 2 / 1e10;
  if (seconds < 1)        return "instantly";
  const units = [["second",1],["minute",60],["hour",3600],
    ["day",86400],["year",31557600],["century",3155760000]];
  for (let i = units.length - 1; i >= 0; i--){
    const n = seconds / units[i][1];
    if (n >= 1) return Math.round(n).toLocaleString() + " " + units[i][0] + (n>=2?"s":"");
  }
  return Math.round(seconds) + " seconds";
}
// > a few centuries? just say "centuries" — the exact number is noise

    Actionable tips + render the bar
    function tips(pw){
  const t = [];
  if (pw.length < 12) t.push("Add length — aim for 12+ characters.");
  if (!/[^a-zA-Z0-9]/.test(pw)) t.push("Mix in a symbol or two.");
  if (/(.)\1{2,}/.test(pw)) t.push("Avoid repeated characters.");
  if (/qwerty|asdf|1234|abcd/i.test(pw)) t.push("Avoid keyboard runs like 'qwerty'.");
  return t;
}
const LABELS = ["Very weak","Weak","Fair","Strong","Very strong"];
function render(pw){
  const { bits, score } = strength(pw);
  segments.forEach((el,i) =>
    el.className = "seg" + (pw && i <= score ? " seg-" + score : ""));
  labelEl.textContent = pw ? LABELS[score] : "—";
  crackEl.textContent = "crack time: " + (pw ? crackTime(bits) : "—");
}
