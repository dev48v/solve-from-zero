    Build a word-frequency table (drop stopwords)
    
        
    
        
    const STOPWORDS = new Set(["the","a","an","is","it","in","on",
  "at","to","and","or","but","of","for","with","this","that",
  "are","was","were","be","been","has","have","had","not","by"]);

function wordFreq(sentences) {
  const freq = {};
  for (const s of sentences) {
    for (const w of tokenize(s)) {
      if (!STOPWORDS.has(w)) freq[w] = (freq[w] || 0) + 1;
    }
  }
  return freq;
}

    Score each sentence by TF
    
        
    
        
    function scoreSentence(sentence, freq) {
  const words = tokenize(sentence).filter(w => !STOPWORDS.has(w));
  if (!words.length) return 0;
  const sum = words.reduce((acc, w) => acc + (freq[w] || 0), 0);
  return sum / words.length;  // normalise by sentence length
}

    Pick top N — but keep original order
    
        
    
        
    function topN(sentences, scores, n) {
  // rank by score → pick indices of top N
  const ranked = scores
    .map((score, i) => ({ i, score }))
    .sort((a, b) => b.score - a.score)
    .slice(0, n)
    .map(x => x.i)
    .sort((a, b) => a - b);  // restore reading order!
  return ranked.map(i => sentences[i]);
}

    function topKeywords(freq, k = 10) {
  return Object.entries(freq)
    .sort((a, b) => b[1] - a[1])
    .slice(0, k)
    .map(([word]) => word);
}

    document.getElementById("go").onclick = () => {
  const text = document.getElementById("inp").value;
  const n    = +document.getElementById("topN").value;
  const sents  = splitSentences(text);
  const freq   = wordFreq(sents);
  const scores = sents.map(s => scoreSentence(s, freq));
  const summary = topN(sents, scores, n);
  const keywords = topKeywords(freq);
  renderResults(summary, keywords, sents.length);
};
