  4. 4
    Find the widest colour channel in a box
    function widestChannel(box) {
  let rMin=255,rMax=0,gMin=255,gMax=0,bMin=255,bMax=0;
  for (const [r,g,b] of box) {
    rMin=Math.min(rMin,r); rMax=Math.max(rMax,r);
    gMin=Math.min(gMin,g); gMax=Math.max(gMax,g);
    bMin=Math.min(bMin,b); bMax=Math.max(bMax,b);
  }
  const rR=rMax-rMin, gR=gMax-gMin, bR=bMax-bMin;
  const m = Math.max(rR,gR,bR);
  return m===rR ? 0 : m===gR ? 1 : 2;   // 0=R 1=G 2=B
}
    5. 
      
  5. 5
    Median-cut: split the biggest box, repeat
    function medianCut(pixels, n) {
  let boxes = [pixels];
  while (boxes.length < n) {
    // split the box with the largest single pixel count
    boxes.sort((a,b) => b.length - a.length);
    const box = boxes.shift();
    const ch = widestChannel(box);
    box.sort((p,q) => p[ch] - q[ch]);     // sort along widest axis
    const mid = box.length >> 1;          // median split
    boxes.push(box.slice(0, mid), box.slice(mid));
  }
  return boxes;
}

  6. 6
    Average each box → one colour
    const palette = boxes.map(box => {
  let r=0,g=0,b=0;
  for (const p of box) { r+=p[0]; g+=p[1]; b+=p[2]; }
  const n = box.length;
  return [Math.round(r/n), Math.round(g/n), Math.round(b/n)];
});
