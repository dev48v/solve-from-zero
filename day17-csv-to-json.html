🔨 Build it yourself

    
Walk the text character-by-character with an inQuotes flag to split into rows and fields safely → turn rows into objects using the header → JSON.stringify → and reverse it: collect keys, quote only the fields that need it. ~60 lines for a correct, library-free converter both ways.

    
    
      
  1. 1
    The naive split — and why it breaks
    // looks fine until a value contains the delimiter:
const line = '1,"Smith, John",30';
line.split(",");
// ➜ ['1', '"Smith', ' John"', '30']   ❌ 4 fields, should be 3
// the comma INSIDE the quotes was treated as a separator.
// you cannot fix this with a regex — you must track state.
    2. 

      
  2. 2
    A stateful char-by-char parser
    // walk every character; remember if we're inside quotes.
function parseCSV(text, delim = ",") {
  const rows = []; let row = [], field = "", inQuotes = false;
  for (let i = 0; i < text.length; i++) {
    const c = text[i], next = text[i + 1];
    if (inQuotes) {
      if (c === '"' && next === '"') { field += '"'; i++; }   // "" → "
      else if (c === '"') inQuotes = false;                  // close quote
      else field += c;                                       // incl. , and \n
    } else {
      if (c === '"') inQuotes = true;                        // open quote
      else if (c === delim) { row.push(field); field = ""; } // field break
      else if (c === "\n") { row.push(field); rows.push(row); row = []; field = ""; }
      else if (c !== "\r") field += c;                       // ignore CR
    }
  }
  row.push(field); rows.push(row);          // flush the last field/row
  return rows;
}
    3. 

      
  3. 3
    Clean up: trim & drop blank lines
    // after parsing, a trailing newline leaves an empty row;
// and stray blank lines shouldn't become {}.
function clean(rows) {
  return rows.filter(r =>
    !(r.length === 1 && r[0].trim() === "")   // skip fully-blank rows
  ).map(r => r.map(cell => cell.trim()));     // trim each cell
}
// note: we trim the PARSED cell — never trim before parsing,
// or you'd corrupt spaces that live inside quoted fields.
    4. 

      
  4. 4
    Rows → objects using the header
    // first row names the columns; the rest become objects.
function rowsToObjects(rows, hasHeader = true) {
  if (rows.length === 0) return [];
  const header = hasHeader
    ? rows[0]
    : rows[0].map((_, i) => "col" + (i + 1));   // synthesize keys
  const body = hasHeader ? rows.slice(1) : rows;
  return body.map(r => {
    const obj = {};
    header.forEach((key, i) => obj[key] = coerce(r[i] ?? ""));
    return obj;
  });
}
    5. 

      
  5. 5
    Type coercion (everything is a string)
    // CSV has no types — "30" is text. Optionally coerce.
function coerce(v) {
  if (v === "") return "";
  if (v === "true")  return true;
  if (v === "false") return false;
  // only treat as a number if it round-trips exactly
  // (so "007" or "1,2" stay strings)
  if (/^-?\d+(\.\d+)?$/.test(v)) {
    const n = Number(v);
    if (String(n) === v) return n;
  }
  return v;
}
    6. 

      
  6. 6
    CSV → JSON, end to end
    function csvToJson(text, { delim = ",", hasHeader = true } = {}) {
  const rows = clean(parseCSV(text, delim));
  const objects = rowsToObjects(rows, hasHeader);
  return JSON.stringify(objects, null, 2);   // pretty-printed
}

csvToJson('id,name,age\n1,"Smith, John",30');
// ➜ [ { "id": 1, "name": "Smith, John", "age": 30 } ]

  7. 7
    Reverse: quote a field only if it needs it
    // a field needs quoting if it contains the delimiter,
// a quote, or a newline. quotes inside get doubled.
function escapeField(value, delim) {
  const s = String(value ?? "");
  if (s.includes(delim) || s.includes('"') ||
      s.includes("\n") || s.includes("\r")) {
    return '"' + s.replace(/"/g, '""') + '"';   // " → ""
  }
  return s;
}
escapeField('Smith, John', ',');  // ➜ "Smith, John"
escapeField('say "hi"',     ',');  // ➜ "say ""hi"""

  8. 8
    JSON → CSV (collect every key)
    function jsonToCsv(json, delim = ",") {
  const arr = JSON.parse(json);                 // throws on bad JSON
  if (!Array.isArray(arr) || arr.length === 0) return "";
  // union of keys across ALL rows (objects may differ)
  const keys = [...new Set(arr.flatMap(o => Object.keys(o)))];
  const head = keys.map(k => escapeField(k, delim)).join(delim);
  const body = arr.map(o =>
    keys.map(k => escapeField(o[k] ?? "", delim)).join(delim)
  );
  return [head, ...body].join("\n");
}
