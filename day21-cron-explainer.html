⏰ Cron Explainer — Read Any Schedule

Type a cron expression — read exactly when it runs.

A cron expression is just five numbers separated by spaces — minute hour day-of-month month day-of-week. Each field says "which values match", using numbers, * (any), , (list), - (range) and / (step). This parses it, describes it in plain English, and computes the next real run times in your browser — nothing is uploaded, no library used.

Cron expression (5 fields)
Coloured fields
…
minute hour day-of-month month day-of-week
🛈 The fields are always in this order: minute, hour, day-of-month, month, day-of-week. Minutes come first — the most common mistake is putting the hour first.
In plain English
…
Next 5 run times — from now, your local time
These are computed by stepping forward minute by minute from now and keeping the first five that match all five fields — the same thing a real cron daemon does every minute.
Field-by-field breakdown
Each field lists the values it matches. * means "every value". A field with a list like 1,15, a range like 1-5, or a step like */15 expands to the exact set shown.

How the parsing works: we split the expression on spaces into exactly five fields, then expand each field into the concrete set of numbers it allows — * becomes the whole range, a-b becomes every number from a to b, */n or a-b/n keeps every n-th value, and commas union several of those together. A given minute "matches" when its minute, hour and month all fall in their sets — and its day matches by the special day-of-month OR day-of-week rule. To list the next runs we start at the next whole minute and walk forward, testing each minute until we've collected five hits.

🔨 Build it yourself

    
Split the expression into five fields → expand each field into the set of numbers it matches → describe it in English → find the next runs by stepping forward one minute at a time and keeping the minutes that match. ~50 lines for a correct, library-free cron parser and scheduler. The one subtlety to get right is the day-of-month OR day-of-week rule.

    Split into exactly five fields
    // a standard cron line is 5 fields separated by whitespace.
const fields = expr.trim().split(/\s+/);
if (fields.length !== 5)
  throw new Error(`Expected 5 fields, got ${fields.length}`);

const [min, hour, dom, mon, dow] = fields;
// order is fixed: minute hour day-of-month month day-of-week

  2. 2
    Know each field's legal range
    // every field has a fixed min..max:
const RANGES = {
  minute:      [0, 59],
  hour:        [0, 23],
  dayOfMonth:  [1, 31],
  month:       [1, 12],
  dayOfWeek:   [0, 6],   // 0 = Sunday … 6 = Saturday
};
// (7 is sometimes also Sunday; we normalise 7 → 0.)
    3. 

      
  3. 3
    Expand one field into a set
    // turn a field string into the exact numbers it matches.
function expand(field, lo, hi){
  const out = new Set();
  for (const part of field.split(",")){          // lists: a,b,c
    let [rangePart, stepStr] = part.split("/");   // steps: x/n
    const step = stepStr ? parseInt(stepStr, 10) : 1;
    let start = lo, end = hi;
    if (rangePart !== "*"){
      const m = rangePart.split("-");             // ranges: a-b
      start = parseInt(m[0], 10);
      end   = m[1] !== undefined ? parseInt(m[1], 10) : start;
    }
    for (let v = start; v <= end; v += step) out.add(v);
  }
  return out;   // e.g. "*/15" over 0..59 → {0,15,30,45}
}
    4. 

      
  4. 4
    Validate as you expand
    // reject anything outside the legal range, or non-numeric.
function expandChecked(field, lo, hi, name){
  const set = expand(field, lo, hi);
  for (const v of set){
    if (Number.isNaN(v) || v < lo || v > hi)
      throw new Error(`${name}: ${v} out of range ${lo}-${hi}`);
  }
  if (set.size === 0) throw new Error(`${name}: matches nothing`);
  return set;
}
    5. 

      
  5. 5
    Does a given minute match?
    // a Date matches when minute, hour and month are all in-set,
// AND the day matches by the special OR rule (see step 6).
function matches(date, sets){
  if (!sets.minute.has(date.getMinutes())) return false;
  if (!sets.hour.has(date.getHours()))     return false;
  if (!sets.month.has(date.getMonth() + 1)) return false; // JS month 0-based!
  return dayMatches(date, sets);
}
    6. 

      
  6. 6
    The day-of-month OR day-of-week rule
    // THE classic cron gotcha. If BOTH day fields are restricted
// (neither is "*"), the day matches when EITHER one matches —
// they are OR'd, not AND'd. If one is "*", only the other counts.
function dayMatches(date, sets){
  const domStar = sets.domRaw === "*";
  const dowStar = sets.dowRaw === "*";
  const domHit = sets.dayOfMonth.has(date.getDate());
  const dowHit = sets.dayOfWeek.has(date.getDay());   // 0=Sun
  if (domStar && dowStar) return true;
  if (domStar) return dowHit;      // only day-of-week set
  if (dowStar) return domHit;      // only day-of-month set
  return domHit || dowHit;         // both set → OR them
}
    7. 

      
  7. 7
    Find the next N runs by minute-stepping
    // start at the NEXT whole minute, then walk forward.
function nextRuns(sets, count){
  const out = [];
  const d = new Date();
  d.setSeconds(0, 0);
  d.setMinutes(d.getMinutes() + 1);          // start from next minute
  // safety cap: 4 years of minutes so we never loop forever
  for (let i = 0; i < 366 * 4 * 24 * 60 && out.length < count; i++){
    if (matches(d, sets)) out.push(new Date(d));
    d.setMinutes(d.getMinutes() + 1);        // Date rolls hours/days/years
  }
  return out;
}
    8. 

      
  8. 8
    Describe one field in English
    // turn a field back into words for the human summary.
function describeField(raw, unit, names){
  if (raw === "*") return `every ${unit}`;
  if (/^\*\/\d+$/.test(raw))
    return `every ${raw.split("/")[1]} ${unit}s`;
  if (/^\d+-\d+$/.test(raw)){
    const [a, b] = raw.split("-");
    return `from ${names ? names[a] : a} through ${names ? names[b] : b}`;
  }
  if (raw.includes(","))
    return raw.split(",").map(v => names ? names[v] : v).join(", ");
  return names ? names[raw] : raw;   // single value
}
    9. 

      
  9. 9
    Stitch the sentence together
    // combine time-of-day + day + month into one readable line.
function describe(min, hour, dom, mon, dow){
  const time = (min !== "*" && hour !== "*")
    ? `at ${pad(hour)}:${pad(min)}`
    : `at minute ${describeField(min,"minute")}, ${describeField(hour,"hour")}`;
  const days = (dow !== "*")
    ? `on ${describeField(dow,"day", DAY_NAMES)}`
    : (dom !== "*" ? `on day-of-month ${describeField(dom,"day")}` : "every day");
  const months = (mon !== "*")
    ? `in ${describeField(mon,"month", MONTH_NAMES)}` : "";
  return [time, days, months].filter(Boolean).join(" ");
  // "0 9 * * 1-5" → "at 09:00 on Mon through Fri"
}
    10. 

      
  10. 10
    Wire it together
    function explainCron(expr){
  const [min, hour, dom, mon, dow] = expr.trim().split(/\s+/);
  const sets = {
    minute:     expandChecked(min,  0, 59, "minute"),
    hour:       expandChecked(hour, 0, 23, "hour"),
    dayOfMonth: expandChecked(dom,  1, 31, "day-of-month"),
    month:      expandChecked(mon,  1, 12, "month"),
    dayOfWeek:  expandChecked(dow,  0, 6,  "day-of-week"),
    domRaw: dom, dowRaw: dow,
  };
  return {
    english: describe(min, hour, dom, mon, dow),
    next:    nextRuns(sets, 5),
  };
}
explainCron("*/15 * * * *");
// ➜ { english:"every 15 minutes", next:[Date, Date, …] }
