Fine print / captions at 12px — the hardest case to pass.

How the number is built: each color's R, G, B is turned into a fraction 0–1, linearized with the sRGB gamma curve (screens store color non-linearly — we undo that), then combined with the weights 0.2126·R + 0.7152·G + 0.0722·B — green counts most because the eye is most sensitive to it. That gives each color a relative luminance L. The ratio is (Llighter + 0.05) / (Ldarker + 0.05); the +0.05 models a little ambient screen glare so pure black never divides by zero.

🔨 Build it yourself

    
Parse the hex into R, G, B → linearize each channel through the sRGB gamma curve → combine with the luminance weights → take the ratio of the two luminances with the +0.05 offset → compare against the WCAG thresholds. ~40 lines for a correct, library-free checker.

    
    
      
  1. 1
    Hex → RGB
    // "#1F2937" → { r: 31, g: 41, b: 55 }
function hexToRgb(hex){
  hex = hex.replace("#", "").trim();
  if (hex.length === 3)                        // shorthand "f80" → "ff8800"
    hex = hex.split("").map(c => c + c).join("");
  const n = parseInt(hex, 16);
  return { r: (n >> 16) & 255, g: (n >> 8) & 255, b: n & 255 };
}
    2. 

      
  2. 2
    Linearize one channel (undo sRGB gamma)
    // screens store color on a curve; undo it before any math.
// input 0–255 → fraction 0–1 → linear light 0–1.
function linearize(channel){
  const c = channel / 255;
  return c <= 0.03928
    ? c / 12.92                              // tiny dark values: straight line
    : Math.pow((c + 0.055) / 1.055, 2.4);   // the rest: power curve
}
    3. 

      
  3. 3
    Relative luminance (weight the channels)
    // combine the three linear channels into one brightness value.
// green weighs most — the human eye is most sensitive to it.
function luminance({ r, g, b }){
  return 0.2126 * linearize(r)
       + 0.7152 * linearize(g)
       + 0.0722 * linearize(b);
}
// luminance is 0 for black, 1 for white.
    4. 

      
  4. 4
    The contrast ratio
    // lighter over darker, each nudged by +0.05 (ambient glare).
function contrastRatio(hexA, hexB){
  const L1 = luminance(hexToRgb(hexA));
  const L2 = luminance(hexToRgb(hexB));
  const lighter = Math.max(L1, L2);
  const darker  = Math.min(L1, L2);
  return (lighter + 0.05) / (darker + 0.05);
}
contrastRatio("#000000", "#FFFFFF");  // ➜ 21
contrastRatio("#777777", "#FFFFFF");  // ➜ 4.48
    5. 

      
  5. 5
    Check it against the WCAG thresholds
    // the four conformance levels, each a simple comparison.
function wcag(ratio){
  return {
    AA_normal:  ratio >= 4.5,   // body text
    AA_large:   ratio >= 3.0,   // ≥18pt, or ≥14pt bold
    AAA_normal: ratio >= 7.0,   // stricter body text
    AAA_large:  ratio >= 4.5,
  };
}
wcag(4.48);  // ➜ { AA_normal:false, AA_large:true, ... }

  7. 7
    Suggest a fix — nudge toward AA
    // if it fails, darken the darker color until it passes 4.5.
// (one easy strategy; you could lighten the light one instead.)
function nudgeToAA(fgHex, bgHex){
  let fg = hexToRgb(fgHex), bg = hexToRgb(bgHex);
  const darken = luminance(fg) < luminance(bg) ? "fg" : "bg";
  let target = darken === "fg" ? fg : bg;
  while (contrastRatio(rgbToHex(fg), rgbToHex(bg)) < 4.5
         && (target.r || target.g || target.b)){
    target.r = Math.max(0, target.r - 8);
    target.g = Math.max(0, target.g - 8);
    target.b = Math.max(0, target.b - 8);
  }
  return { fg: rgbToHex(fg), bg: rgbToHex(bg) };
}
    8. 

      
  8. 8
    Wire it all together
    function update(){
  const fg = fgHex.value, bg = bgHex.value;
  const ratio = contrastRatio(fg, bg);
  ratioEl.textContent = ratio.toFixed(2).replace(/\.00$/, "") + ":1";
  const r = wcag(ratio);
  setBadge(badgeAA,  r.AA_normal);
  setBadge(badgeAAl, r.AA_large);
  // ...and paint the live preview
  preview.style.color = fg;
  preview.style.background = bg;
}
update();   // run once on load — done.
