🔨 Build it yourself

    
Parse the text in a try/catch, pretty-print with JSON.stringify(obj, null, 2), tokenize the result into coloured spans, offer a minify, and turn the parser's error into a line/column you can point at. ~60 lines for a real JSON formatter & validator.

    
    
      
  1. 1
    Parse the input safely with try/catch
    const text = inputEl.value;
let value = null, errorMsg = "";
try {
  // JSON.parse turns the TEXT into a real JS value (object/array/…).
  // Any syntax problem THROWS a SyntaxError here — we catch it.
  value = JSON.parse(text);
} catch (err) {
  errorMsg = err.message;   // e.g. "Unexpected token } in JSON at position 42"
}
// whether parse threw IS the validation result — nothing else needed

    Syntax-highlight: tokenize, then wrap in spans
    function highlight(json) {
  // 1) escape HTML so the JSON can't inject markup
  json = json.replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;");
  // 2) one regex finds strings (incl. keys), numbers, booleans, null
  return json.replace(
    /("(\\u[a-zA-Z0-9]{4}|\\[^u]|[^\\"])*"(\s*:)?|\b(true|false|null)\b|-?\d+(?:\.\d+)?(?:[eE][+-]?\d+)?)/g,
    (match) => {
      let cls = "tok-num";
      if (/^"/.test(match)) cls = /:$/.test(match) ? "tok-key" : "tok-str";
      else if (/true|false/.test(match)) cls = "tok-bool";
      else if (/null/.test(match)) cls = "tok-null";
      return `<span class="${cls}">${match}</span>`;
    });
}
outputEl.innerHTML = highlight(pretty);

  5. 5
    Turn the error into a line & column
    // V8's message ends with "...at position N"; Firefox gives "line L column C"
function locate(text, msg) {
  const pos = msg.match(/position (\d+)/);
  if (pos) {
    const idx = +pos[1];
    const before = text.slice(0, idx);
    const line = before.split("\n").length;            // 1-based line
    const col  = idx - before.lastIndexOf("\n");        // 1-based column
    return { line, col, idx };
  }
  const lc = msg.match(/line (\d+) column (\d+)/);
  if (lc) return { line:+lc[1], col:+lc[2] };
  return null;
}

  6. 6
    Point at the offending snippet
    // show the bad line and a caret ^ under the exact column
function pointAt(text, loc) {
  const lines = text.split("\n");
  const bad = lines[loc.line - 1] || "";
  const caret = " ".repeat(Math.max(0, loc.col - 1)) + "^";
  return `line ${loc.line}, column ${loc.col}\n` +
         bad + "\n" + caret;
}
// → exactly where to fix it, instead of a vague "invalid JSON"
